import { Injectable, BadRequestException, NotFoundException } from '@nestjs/common';
import { PrismaService } from '../../prisma/prisma.service';
import { EncryptionService } from '../../providers/encryption/encryption.service';
import { BlockchainService } from '../../providers/blockchain/blockchain.service';
import * as bip39 from 'bip39';
import { ethers } from 'ethers';
import * as bitcoin from 'bitcoinjs-lib';
import { Keypair } from '@solana/web3.js';
import * as bip32 from 'bip32';
import * as bs58 from 'bs58';
import { CreateWalletDto } from './dto/create-wallet.dto';
import { ImportWalletDto } from './dto/import-wallet.dto';

@Injectable()
export class WalletService {
  constructor(
    private prisma: PrismaService,
    private encryptionService: EncryptionService,
    private blockchainService: BlockchainService,
  ) {}

  /**
   * Generate a new HD wallet with mnemonic phrase
   */
  async generateWallet(userId: string, dto: CreateWalletDto) {
    // Generate mnemonic (12 or 24 words)
    const mnemonic = bip39.generateMnemonic(dto.strength || 256);
    
    // Validate mnemonic
    if (!bip39.validateMnemonic(mnemonic)) {
      throw new BadRequestException('Invalid mnemonic generated');
    }

    // Generate seed from mnemonic
    const seed = await bip39.mnemonicToSeed(mnemonic);
    const seedHex = seed.toString('hex');

    // Derive wallet based on chain
    const walletData = await this.deriveWalletFromSeed(
      seedHex,
      dto.chain,
      dto.path || this.getDefaultPath(dto.chain),
    );

    // Encrypt private key
    const encryptedPrivateKey = await this.encryptionService.encrypt(
      walletData.privateKey,
    );

    // Hash mnemonic for verification (never store plaintext)
    const mnemonicHash = this.encryptionService.hashMnemonic(mnemonic);

    // Create wallet record
    const wallet = await this.prisma.wallet.create({
      data: {
        userId,
        name: dto.name || `${dto.chain} Wallet`,
        type: 'HD',
        chain: dto.chain,
        address: walletData.address,
        encryptedPrivateKey,
        mnemonicHash,
        path: dto.path || this.getDefaultPath(dto.chain),
      },
    });

    // Return wallet info with mnemonic (only shown once!)
    return {
      wallet: this.sanitizeWallet(wallet),
      mnemonic, // IMPORTANT: Show only once during creation
      warning: 'Store your mnemonic phrase securely. It will never be shown again.',
    };
  }

  /**
   * Import existing wallet by private key or mnemonic
   */
  async importWallet(userId: string, dto: ImportWalletDto) {
    let privateKey: string;
    let mnemonicHash: string | null = null;

    if (dto.mnemonic) {
      // Import from mnemonic
      if (!bip39.validateMnemonic(dto.mnemonic)) {
        throw new BadRequestException('Invalid mnemonic phrase');
      }

      const seed = await bip39.mnemonicToSeed(dto.mnemonic);
      const seedHex = seed.toString('hex');
      
      const walletData = await this.deriveWalletFromSeed(
        seedHex,
        dto.chain,
        dto.path || this.getDefaultPath(dto.chain),
      );

      privateKey = walletData.privateKey;
      mnemonicHash = this.encryptionService.hashMnemonic(dto.mnemonic);
      
      // Verify address matches if provided
      if (dto.address && walletData.address !== dto.address) {
        throw new BadRequestException('Derived address does not match provided address');
      }
    } else if (dto.privateKey) {
      // Import from private key
      privateKey = dto.privateKey;
      
      // Derive address from private key
      const wallet = new ethers.Wallet(privateKey);
      if (dto.address && wallet.address !== dto.address) {
        throw new BadRequestException('Private key does not match provided address');
      }
    } else {
      throw new BadRequestException('Must provide either mnemonic or private key');
    }

    // Encrypt private key
    const encryptedPrivateKey = await this.encryptionService.encrypt(privateKey);

    // Get address from private key
    const address = await this.getAddressFromPrivateKey(privateKey, dto.chain);

    // Create wallet record
    const wallet = await this.prisma.wallet.create({
      data: {
        userId,
        name: dto.name || `Imported ${dto.chain} Wallet`,
        type: dto.type || 'IMPORTED',
        chain: dto.chain,
        address,
        encryptedPrivateKey,
        mnemonicHash,
        path: dto.path,
      },
    });

    return {
      wallet: this.sanitizeWallet(wallet),
      message: 'Wallet imported successfully',
    };
  }

  /**
   * Create watch-only wallet (no private key)
   */
  async createWatchWallet(userId: string, address: string, chain: string, name?: string) {
    // Validate address format
    if (!this.blockchainService.isValidAddress(address, chain)) {
      throw new BadRequestException(`Invalid ${chain} address format`);
    }

    const wallet = await this.prisma.wallet.create({
      data: {
        userId,
        name: name || `Watch ${chain} Wallet`,
        type: 'WATCH_ONLY',
        chain: chain as any,
        address,
      },
    });

    return this.sanitizeWallet(wallet);
  }

  /**
   * Get wallet by ID
   */
  async getWallet(walletId: string, userId: string) {
    const wallet = await this.prisma.wallet.findFirst({
      where: { id: walletId, userId, deletedAt: null },
      include: {
        tokens: true,
        transactions: {
          orderBy: { createdAt: 'desc' },
          take: 50,
        },
      },
    });

    if (!wallet) {
      throw new NotFoundException('Wallet not found');
    }

    return this.sanitizeWallet(wallet);
  }

  /**
   * List all wallets for user
   */
  async listWallets(userId: string, chain?: string) {
    const wallets = await this.prisma.wallet.findMany({
      where: {
        userId,
        deletedAt: null,
        ...(chain && { chain: chain as any }),
      },
      include: {
        tokens: true,
      },
      orderBy: { createdAt: 'desc' },
    });

    return wallets.map(w => this.sanitizeWallet(w));
  }

  /**
   * Delete wallet (soft delete)
   */
  async deleteWallet(walletId: string, userId: string) {
    const wallet = await this.prisma.wallet.findFirst({
      where: { id: walletId, userId },
    });

    if (!wallet) {
      throw new NotFoundException('Wallet not found');
    }

    await this.prisma.wallet.update({
      where: { id: walletId },
      data: { deletedAt: new Date() },
    });

    return { message: 'Wallet deleted successfully' };
  }

  /**
   * Export wallet (returns encrypted private key)
   */
  async exportWallet(walletId: string, userId: string, password: string) {
    const wallet = await this.prisma.wallet.findFirst({
      where: { id: walletId, userId, deletedAt: null },
    });

    if (!wallet) {
      throw new NotFoundException('Wallet not found');
    }

    if (!wallet.encryptedPrivateKey) {
      throw new BadRequestException('This wallet has no private key to export');
    }

    // Verify user password before exporting
    // This is an additional security layer
    const decryptedKey = await this.encryptionService.decrypt(
      wallet.encryptedPrivateKey,
    );

    // Re-encrypt with user's export password
    const exportEncryptedKey = await this.encryptionService.encryptWithPassword(
      decryptedKey,
      password,
    );

    return {
      encryptedPrivateKey: exportEncryptedKey,
      address: wallet.address,
      message: 'Keep this encrypted key secure. Do not share it with anyone.',
    };
  }

  /**
   * Derive wallet from seed phrase
   */
  private async deriveWalletFromSeed(
    seedHex: string,
    chain: string,
    path: string,
  ): Promise<{ address: string; privateKey: string; publicKey: string }> {
    switch (chain.toUpperCase()) {
      case 'ETHEREUM':
      case 'POLYGON':
      case 'BSC':
      case 'AVALANCHE':
      case 'ARBITRUM':
      case 'OPTIMISM':
        return this.deriveEthereumWallet(seedHex, path);
      
      case 'BITCOIN':
      case 'LITECOIN':
        return this.deriveBitcoinWallet(seedHex, path, chain);
      
      case 'SOLANA':
        return this.deriveSolanaWallet(seedHex, path);
      
      default:
        throw new BadRequestException(`Unsupported chain: ${chain}`);
    }
  }

  /**
   * Derive Ethereum-compatible wallet
   */
  private deriveEthereumWallet(seedHex: string, path: string) {
    const hdNode = ethers.HDNodeWallet.fromSeed(seedHex);
    const derivedNode = hdNode.derivePath(path);
    
    return {
      address: derivedNode.address,
      privateKey: derivedNode.privateKey,
      publicKey: derivedNode.publicKey,
    };
  }

  /**
   * Derive Bitcoin wallet
   */
  private deriveBitcoinWallet(seedHex: string, path: string, chain: string) {
    const network = chain === 'LITECOIN' 
      ? bitcoin.networks.litecoin 
      : bitcoin.networks.bitcoin;
    
    const root = bip32.fromSeed(Buffer.from(seedHex, 'hex'), network);
    const child = root.derivePath(path);
    
    const { address } = bitcoin.payments.p2pkh({
      pubkey: Buffer.from(child.publicKey),
      network,
    });

    return {
      address: address!,
      privateKey: child.toWIF(),
      publicKey: child.publicKey.toString('hex'),
    };
  }

  /**
   * Derive Solana wallet
   */
  private deriveSolanaWallet(seedHex: string, path: string) {
    // Solana uses ed25519 with a specific derivation
    const seed = Buffer.from(seedHex, 'hex');
    const keypair = Keypair.fromSeed(seed.slice(0, 32));
    
    return {
      address: keypair.publicKey.toString(),
      privateKey: bs58.encode(keypair.secretKey),
      publicKey: keypair.publicKey.toString(),
    };
  }

  /**
   * Get address from private key
   */
  private async getAddressFromPrivateKey(privateKey: string, chain: string): Promise<string> {
    switch (chain.toUpperCase()) {
      case 'ETHEREUM':
      case 'POLYGON':
      case 'BSC':
        const wallet = new ethers.Wallet(privateKey);
        return wallet.address;
      
      case 'SOLANA':
        const secretKey = bs58.decode(privateKey);
        const keypair = Keypair.fromSecretKey(secretKey);
        return keypair.publicKey.toString();
      
      default:
        throw new BadRequestException(`Unsupported chain: ${chain}`);
    }
  }

  /**
   * Get default BIP44 derivation path for chain
   */
  private getDefaultPath(chain: string): string {
    const paths = {
      ETHEREUM: "m/44'/60'/0'/0/0",
      POLYGON: "m/44'/60'/0'/0/0",
      BSC: "m/44'/60'/0'/0/0",
      BITCOIN: "m/44'/0'/0'/0/0",
      LITECOIN: "m/44'/2'/0'/0/0",
      SOLANA: "m/44'/501'/0'/0'",
    };
    return paths[chain.toUpperCase()] || "m/44'/60'/0'/0/0";
  }

  /**
   * Sanitize wallet for response (remove sensitive data)
   */
  private sanitizeWallet(wallet: any) {
    const { encryptedPrivateKey, mnemonicHash, ...safe } = wallet;
    return safe;
  }
}

import { JwtService } from '@nestjs/jwt';
import { PrismaService } from '../../prisma/prisma.service';
import { ethers } from 'ethers';
import { v4 as uuidv4 } from 'uuid';
import * as nacl from 'tweetnacl';
import * as bs58 from 'bs58';
import { SiweMessage } from 'siwe';

@Injectable()
export class WalletAuthService {
  constructor(
    private prisma: PrismaService,
    private jwtService: JwtService,
  ) {}

  /**
   * Generate nonce for wallet authentication
   */
  async generateNonce(walletAddress: string) {
    const nonce = uuidv4();
    const message = this.createSignInMessage(walletAddress, nonce);
    
    const expiresAt = new Date(Date.now() + 10 * 60 * 1000); // 10 minutes

    const wallet = await this.prisma.wallet.findUnique({
      where: { address: walletAddress },
    });

    if (!wallet) {
      throw new BadRequestException('Wallet not registered. Please create wallet first.');
    }

    await this.prisma.walletAuth.create({
      data: {
        walletId: wallet.id,
        nonce,
        message,
        expiresAt,
      },
    });

    return { nonce, message, expiresAt };
  }

  /**
   * Verify signature and authenticate wallet
   */
  async verifySignature(
    walletAddress: string,
    signature: string,
    nonce: string,
    chain: string = 'ETHEREUM',
  ) {
    const authRecord = await this.prisma.walletAuth.findFirst({
      where: {
        wallet: { address: walletAddress },
        nonce,
        usedAt: null,
        expiresAt: { gt: new Date() },
      },
      include: { wallet: true },
    });

    if (!authRecord) {
      throw new UnauthorizedException('Invalid or expired nonce');
    }

    let isValid = false;

    switch (chain.toUpperCase()) {
      case 'ETHEREUM':
      case 'POLYGON':
      case 'BSC':
        isValid = await this.verifyEthereumSignature(
          authRecord.message,
          signature,
          walletAddress,
        );
        break;
      
      case 'SOLANA':
        isValid = this.verifySolanaSignature(
          authRecord.message,
          signature,
          walletAddress,
        );
        break;
      
      default:
        throw new BadRequestException(`Unsupported chain for auth: ${chain}`);
    }

    if (!isValid) {
      throw new UnauthorizedException('Invalid signature');
    }

    // Mark nonce as used
    await this.prisma.walletAuth.update({
      where: { id: authRecord.id },
      data: { usedAt: new Date() },
    });

    // Generate JWT token
    const token = this.jwtService.sign({
      sub: authRecord.wallet.userId,
      walletAddress: walletAddress,
      chain,
    });

    return {
      accessToken: token,
      walletAddress,
      chain,
      message: 'Authentication successful',
    };
  }

  /**
   * Verify Ethereum signature (EIP-4361: Sign-In with Ethereum)
   */
  private async verifyEthereumSignature(
    message: string,
    signature: string,
    address: string,
  ): Promise<boolean> {
    try {
      const siweMessage = new SiweMessage(message);
      const result = await siweMessage.verify({ signature });
      return result.success && result.data.address.toLowerCase() === address.toLowerCase();
    } catch (error) {
      // Fallback to basic verification
      try {
        const recoveredAddress = ethers.verifyMessage(message, signature);
        return recoveredAddress.toLowerCase() === address.toLowerCase();
      } catch {
        return false;
      }
    }
  }

  /**
   * Verify Solana signature
   */
  private verifySolanaSignature(
    message: string,
    signature: string,
    address: string,
  ): boolean {
    try {
      const messageBytes = new TextEncoder().encode(message);
      const signatureBytes = bs58.decode(signature);
      const publicKeyBytes = bs58.decode(address);
      
      return nacl.sign.detached.verify(
        messageBytes,
        signatureBytes,
        publicKeyBytes,
      );
    } catch {
      return false;
    }
  }

  /**
   * Create EIP-4361 sign-in message
   */
  private createSignInMessage(address: string, nonce: string): string {
    const message = new SiweMessage({
      domain: 'taskflow.io',
      address,
      statement: 'Sign in with Ethereum to TaskFlow Pro',
      uri: 'https://taskflow.io',
      version: '1',
      chainId: 1,
      nonce,
      issuedAt: new Date().toISOString(),
    });
    
    return message.prepareMessage();
  }
}

import { Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import * as crypto from 'crypto';

@Injectable()
export class EncryptionService {
  private readonly algorithm = 'aes-256-gcm';
  private readonly keyLength = 32;
  private readonly ivLength = 16;
  private readonly tagLength = 16;
  private masterKey: Buffer;

  constructor(private configService: ConfigService) {
    const keyHex = this.configService.get('ENCRYPTION_KEY');
    if (!keyHex) {
      throw new Error('ENCRYPTION_KEY environment variable is required');
    }
    this.masterKey = Buffer.from(keyHex, 'hex');
  }

  /**
   * Encrypt data with AES-256-GCM
   */
  async encrypt(plaintext: string): Promise<string> {
    const iv = crypto.randomBytes(this.ivLength);
    const cipher = crypto.createCipheriv(this.algorithm, this.masterKey, iv);
    
    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    const authTag = cipher.getAuthTag();
    
    // Combine IV + Auth Tag + Encrypted Data
    const combined = Buffer.concat([
      iv,
      authTag,
      Buffer.from(encrypted, 'hex'),
    ]);
    
    return combined.toString('base64');
  }

  /**
   * Decrypt data with AES-256-GCM
   */
  async decrypt(encryptedData: string): Promise<string> {
    const combined = Buffer.from(encryptedData, 'base64');
    
    const iv = combined.subarray(0, this.ivLength);
    const authTag = combined.subarray(this.ivLength, this.ivLength + this.tagLength);
    const encrypted = combined.subarray(this.ivLength + this.tagLength);
    
    const decipher = crypto.createDecipheriv(this.algorithm, this.masterKey, iv);
    decipher.setAuthTag(authTag);
    
    let decrypted = decipher.update(encrypted.toString('hex'), 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return decrypted;
  }

  /**
   * Encrypt with user password (for export)
   */
  async encryptWithPassword(plaintext: string, password: string): Promise<string> {
    const salt = crypto.randomBytes(32);
    const key = crypto.pbkdf2Sync(password, salt, 100000, 32, 'sha512');
    const iv = crypto.randomBytes(16);
    
    const cipher = crypto.createCipheriv(this.algorithm, key, iv);
    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    const authTag = cipher.getAuthTag();
    
    const combined = Buffer.concat([salt, iv, authTag, Buffer.from(encrypted, 'hex')]);
    return combined.toString('base64');
  }

  /**
   * Hash mnemonic for verification (never store plaintext)
   */
  hashMnemonic(mnemonic: string): string {
    return crypto
      .createHash('sha512')
      .update(mnemonic + this.masterKey.toString('hex'))
      .digest('hex');
  }

  /**
   * Verify mnemonic against stored hash
   */
  verifyMnemonic(mnemonic: string, hash: string): boolean {
    const calculatedHash = this.hashMnemonic(mnemonic);
    return crypto.timingSafeEqual(
      Buffer.from(calculatedHash, 'hex'),
      Buffer.from(hash, 'hex'),
    );
  }

  /**
   * Generate secure random encryption key
   */
  static generateEncryptionKey(): string {
    return crypto.randomBytes(32).toString('hex');
  }
}

=================================================================
TROUBLESHOOTING
=================================================================

1. Database connection issues:
- Check DATABASE_URL in .env
- Ensure PostgreSQL is running
- Check firewall rules

2. RPC connection issues:
- Verify RPC URLs are accessible
- Check rate limits on free RPC providers
- Consider using dedicated node service

3. Transaction failures:
- Check gas price and limit
- Verify wallet has sufficient balance
- Check nonce (try resetting if stuck)

4. Private key decryption errors:
- Verify ENCRYPTION_KEY hasn't changed
- Check if key was rotated without re-encryption
